```sql
SELECT p.name AS product,
       ROUND(SUM(oi.quantity * oi.unitPrice), 2) AS revenue,
       SUM(oi.quantity) AS units_sold,
       ROUND(100.0 * SUM(oi.quantity * oi.unitPrice) / tot.total_revenue, 2) AS pct_share
FROM OrderItem oi
JOIN Product p ON oi.productId = p.id
JOIN "Order" o ON oi.orderId = o.id
CROSS JOIN (
  SELECT COALESCE(SUM(oi2.quantity * oi2.unitPrice),0) AS total_revenue
  FROM OrderItem oi2
  JOIN "Order" o2 ON oi2.orderId = o2.id
  WHERE o2.status != 'Cancelled' AND date(o2.orderDate) BETWEEN date('now','-30 days') AND date('now')
) tot
WHERE o.status != 'Cancelled' AND date(o.orderDate) BETWEEN date('now','-30 days') AND date('now')
GROUP BY p.name
ORDER BY revenue DESC
LIMIT 10;
```
